CREATE TABLE `orare` (
  `id_orar` int NOT NULL AUTO_INCREMENT,
  `nume_orar` varchar(100) NOT NULL,
  PRIMARY KEY (`id_orar`),
  UNIQUE KEY `nume_orar_UNIQUE` (`nume_orar`)
) ENGINE=InnoDB AUTO_INCREMENT=724 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `ore` (
  `id` int NOT NULL AUTO_INCREMENT,
  `id_orar` int NOT NULL,
  `ora_inceput` varchar(45) NOT NULL,
  `ora_final` varchar(45) NOT NULL,
  `profesor` varchar(45) DEFAULT NULL,
  `materie` varchar(45) NOT NULL,
  `sala` varchar(45) DEFAULT NULL,
  `saptamana` varchar(45) NOT NULL,
  `grupa` varchar(45) DEFAULT NULL,
  `zi` varchar(45) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `id_orar_idx` (`id_orar`),
  CONSTRAINT `id_orar` FOREIGN KEY (`id_orar`) REFERENCES `orare` (`id_orar`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=1094 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


CREATE TABLE `warnings` (
  `id_warning` int NOT NULL AUTO_INCREMENT,
  `id_orar` int NOT NULL,
  `warning_details` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`id_warning`),
  KEY `id_orar_idx` (`id_orar`),
  CONSTRAINT `id_orar_warn` FOREIGN KEY (`id_orar`) REFERENCES `orare` (`id_orar`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=125 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
